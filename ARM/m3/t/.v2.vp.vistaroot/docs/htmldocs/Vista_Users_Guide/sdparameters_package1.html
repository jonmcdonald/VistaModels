sdParameters Package 

     #ifndef MentorGraphics__sdParameters__h
     #define MentorGraphics__sdParameters__h
     
     /* 
        This package contains functions for accessing parameters specified
        in a text file.
     
     
        Description:
        ============
     
        1. The following types are supported:
     
           char, signed char, unsigned char,
           short, unsigned short,
           int,   unsigned int,
           long,  unsigned long,
           long long, unsigned long long (not supported in Microsoft Visual Studio)
           int64, uint64,                (defined in SystemC)
           float, double,
           char *, const char *,
           bool,
           sc_time                       (defined in SystemC)
     
        2. To use this package:
               Declare a member in your SystemC module and initialize it 
           Using the SD_INITIALIZE_PARAMETER macro. This macro has 2 
           arguments:
           (1) the name of the parameter itself; and (2) its default value.
           For example:
     
           ...
           SC_MODULE(my_module)
           {
              ...
              int param;     // declares the parameter named "param"
              ...
     
              SC_CTOR(my_module) :
                 ...
                 SD_INITIALIZE_PARAMETER(param, 5) // initialization of 
                                   // param having a default value of 5.
                 ...
                 {
                    ...
                 }
           }
     
           If the parameter value is defined in the parameter file, it is 
           used; otherwise, the default value is used.
     
      3. It is possible to initialize a member of a type other than those
           supported (see above) using the parameter file with the
           D_INITIALIZE_PARAMETER_AS macro. However, to do so, the
           parameter's type has to have the appropriate constructor.
     
           This macro has one additional argument (the first one) that
           defines the supported type used for initialization. This macro 
           can be used for the following SystemC types: sc_int, sc_uint, 
           sc_bv etc.
     
           For example:
     
           ...
           SC_MODULE(my_module)
           {
              ...
              sc_uint<40> param1;     // declares the parameter named 
                                             //"param1"
              sc_uint<16> param2;     // declares the parameter named 
                                             //"param2"
     
              SC_CTOR(my_module) :
                 ...
                             
                             // initialization of param1 of the type 
                                    //sc_uint<40> using the type uint64.
                 SD_INITIALIZE_PARAMETER_AS(uint64, param1, (uint64) 1),
                             
                 
                 // initialization of param2 of the type sc_uint<16> using 
                         //the type int.
                 SD_INITIALIZE_PARAMETER_AS(int,    param2, 1)
                 ...
                 {
                    ...
                 }
           }
     
        4. Following is the set of functions that can be used to obtain the 
           value of any parameter:
     
              sdGetParameterAsInt,
              sdGetParameterAsUInt,
              sdGetParameterAsInt64,
              sdGetParameterAsUInt64,
              sdGetParameterAsDouble,
              sdGetParameterAsString,
              sdGetParameterAsConstString,
              sdGetParameterAsTime,
              sdGetParameterAsBool
     
           These functions have 2 arguments: parameterName and 
           hierInstanceName.
           The second argument may be omitted, in which case the current 
           Instance is used. Note that calling these functions during 
           elaboration with a non-null value for the second parameter 
           (hierInstanceName) will yield unpredictable results.
     
           If a parameter value is declared in the parameter file or it is
           initialized using SD_INITIALIZE_PARAMETER or 
           SD_INITIALIZE_PARAMETER_AS macros, this value is returned.
           Otherwise the "zero" value is returned; where
           "zero" will be one of the following, depending on the returned 
           type: 
           0, 0.0, false, or SC_ZERO_TIME.
     
           To verify that the parameter value is defined, call the 
           sdParameterExists function. This function has the same arguments 
           as those described above, and returns either true (defined) or 
           false (not defined).
     
           Important:
           If you assign the parameter value within a SystemC process or
           within the module's contructor, the new value will not be 
           returned by the above functions. To avoid this problem, use the 
           "const" keyword,  when you declare the module's member field. For
           example:
     
           ...
           const int param;
           ...
     
           When you assign a value to such a member, this will result 
               in a compilation error. A different compilation error will
           be displayed if you forget to initialize the member.
     
               
        The Parameters Text File Format:
        ================================
     
        Any line beginning with the # character is treated as a comment,
        and ignored. (Note that the line must begin with the # character,
        otherwise it is not treated as a comment.)
     
        Any empty line or a line consisting of only whitespace characters
        is ignored.
     
        All the lines that define the parameter values take the form:
     
        instance_path.parameter_name = value
     
        The instance_path part of the LHS is the full SystemC path of a
        component instance in the simulation, starting with the top-level
        block diagram, with each level of the hierarchy separated by the
        "."  character. The name of the parameter is as defined by the
        component model. Names of parameters, as well as the instance
        paths, are case sensitive.
     
        The = character separates the name of the parameter from the
        value. There may or may not be any whitespace around the =
        character. The value of the parameter starts at the first
        non-whitespace character after the = character, and extends to the
        end of the line, excluding any terminating whitespace. Any
        intermediate whitespace is included in the value of the parameter.
     
        The file can specify parameters for multiple designs, since the
        parameters are specified using the full instances names. Unused
        parameters are simply ignored. There is no need to order the
        parameters in any particular way. If the same parameter is
        specified more than once, the last specification will override all
        previous specifications.
     
        There is an alternative syntax that allows all instances of a
        specific component to share a parameter value. This syntax is:
     
        :component_name.parameter_name = value
          or
        library_name:component_name.parameter_name = value (used with VE 
        only)
     
        This syntax, when used, will set the same value for all instances
        of the component in the design, regardless of the instance paths.
     
        Parameter values specified using the full instance path syntax will
        override parameter values specified using the :component or 
        library:component
        syntax, so it is possible to define a "generic" value and override
        it for a specific instance.
     
        The third syntax is:
     
           *instance_path.parameter_name = value
     
        This syntax means, that the value of parameters named 
        "parameter_name", will be set in the "instance_path" instance itself 
        and all its subdesigns.
     
        Look at saParameters.h file (or saParameters package) for more 
        details.
     
     */

     int sdGetParameterAsInt(const char * parameterName,
                             const char * hierInstanceName = 0);
     
     unsigned int sdGetParameterAsUInt(const char * parameterName,
                                       const char * hierInstanceName = 0);
     
     int64 sdGetParameterAsInt64(const char * parameterName,
                                 const char * hierInstanceName = 0);
     
     uint64 sdGetParameterAsUInt64(const char * parameterName,
                                   const char * hierInstanceName = 0);
     
     double sdGetParameterAsDouble(const char * parameterName,
                                   const char * hierInstanceName = 0);
     
     // Caller is responsible to delete the returned value using "delete []"
     char * sdGetParameterAsString(const char * parameterName,
                                   const char * hierInstanceName = 0);
     
     // Caller MUST NOT delete the returned value
     const char * sdGetParameterAsConstString(const char * parameterName,
                                              const char * hierInstanceName = 0);
     sc_time sdGetParameterAsTime(const char * parameterName,
                                  const char * hierInstanceName = 0);
     
     bool sdGetParameterAsBool(const char * parameterName,
                               const char * hierInstanceName = 0);
     
     bool sdParameterExists(const char * parameterName,
                            const char * hierInstanceName = 0);
     
     
     
     // These functions are used internally.
     // The main purpose is to set explicitly the moduleName
     // and the useReverseHierarchy flag.
     
     int sdGetParameterAsInt(const char * parameterName,
                             const char * hierInstanceName,
                             const char * moduleName,
                             bool         useReverseHierarchy = false);
     
     unsigned int sdGetParameterAsUInt(const char * parameterName,
                                       const char * hierInstanceName,
                                       const char * moduleName,
                                       bool         useReverseHierarchy = 
                                                       false);
     
     int64 sdGetParameterAsInt64(const char * parameterName,
                                 const char * hierInstanceName,
                                 const char * moduleName,
                                 bool         useReverseHierarchy = false);
     
     uint64 sdGetParameterAsUInt64(const char * parameterName,
                                   const char * hierInstanceName,
                                   const char * moduleName,
                                   bool         useReverseHierarchy = 
                                                    false);
     
     double sdGetParameterAsDouble(const char * parameterName, 
                                   const char * hierInstanceName,
                                   const char * moduleName,
                                   bool         useReverseHierarchy = 
                                                    false);
     
     // Caller is responsible to delete the returned value using "delete []"
     char * sdGetParameterAsString(const char * parameterName,
                                   const char * hierInstanceName,
                                   const char * moduleName,
                                   bool         useReverseHierarchy = 
                                                     false);
     
     // Caller MUST NOT delete the returned value
     const char * sdGetParameterAsConstString(const char * parameterName,
                                              const char * hierInstanceName,
                                              const char * moduleName,
                                        bool useReverseHierarchy = false);
     
     sc_time sdGetParameterAsTime(const char * parameterName,
                                  const char * hierInstanceName,
                                  const char * moduleName,
                                  bool         useReverseHierarchy = false);
     
     bool sdGetParameterAsBool(const char * parameterName,
                               const char * hierInstanceName,
                               const char * moduleName,
                               bool         useReverseHierarchy = false);
     
     bool sdParameterExists(const char * parameterName,
                            const char * hierInstanceName,
                            const char * moduleName,
                            bool         useReverseHierarchy = false);
     
     const void * sdInitParameterInternal(const char * parameterName,
                                          const char * hierInstanceName,
                                          const char * moduleName,
                                          const void * defaultValue,
                                          const char * typeName);
     
     template <class T>
     inline T sdInitParameter(const char * parameterName,
                              const T    & defaultValue,
                              const char * hierInstanceName = 0,
                              const char * moduleName = 0)
     {
       return * (T *) sdInitParameterInternal(parameterName,
                                              hierInstanceName,
                                              moduleName,
                                              &defaultValue,
                                              typeid(T).name());
     }
     
     
     
     #ifndef VIRTUAL_PROTOTYPE
     
     // to be used with a systemc module constructor
     
     #define SD_INITIALIZE_PARAMETER(parameter, defaultValue) \
        parameter( sdInitParameter<__typeof__(parameter)> ( #parameter, 
           defaultValue, name() ) )
     
     #define SD_INITIALIZE_PARAMETER_AS(type, parameter, defaultValue) \
        parameter( sdInitParameter<type> ( #parameter, defaultValue, name() 
           ) )
     
     // Explicitly set the module and hier instance name. Should not be used 
     // outside of VE.
     
     #define SD_INITIALIZE_PARAMETER_VE(parameter, defaultValue, 
            hierInstanceName, moduleName) \
        parameter( sdInitParameter<__typeof__(parameter)> ( #parameter, 
            defaultValue,  hierInstanceName, moduleName) )
     #define SD_INITIALIZE_PARAMETER_VE_AS(type, parameter, defaultValue, 
            hierInstanceName, moduleName) \
        parameter( sdInitParameter<type>( #parameter, defaultValue, 
            hierInstanceName, moduleName) )
     
     #else
     
     #define SD_INITIALIZE_PARAMETER(parameter, defaultValue) \
        parameter (defaultValue)
     
     #define SD_INITIALIZE_PARAMETER_AS(type, parameter, defaultValue) \
        parameter (defaultValue)
     
     #define SD_INITIALIZE_PARAMETER_VE(parameter, defaultValue, 
             hierInstanceName, moduleName) \
        parameter (defaultValue)
     
     #define SD_INITIALIZE_PARAMETER_VE_AS(type, parameter, defaultValue, 
             hierInstanceName, moduleName) \
        parameter (defaultValue)
     
     #endif
     
     #endif // MentorGraphics__sdParameters__h
